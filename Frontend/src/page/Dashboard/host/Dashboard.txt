import React from "react";
import { useLocation, useNavigate } from "react-router-dom";

const HostDashboard = () => {
  const location = useLocation();
  const navigate = useNavigate();
  
  // Normalize initial user data from navigation state if provided..
  3
  // Accept either { userData: { user: {...} } } or { user: {...} } shapes.
  const initialUser = location.state?.userData ?? (location.state?.user ? { user: location.state.user } : null);
  const [userData, setUserData] = React.useState(initialUser);
  
  // On mount, if we don't have userData, try to recover from session or redirect to login.
  React.useEffect(() => {
    if (!userData) {
      const sessionId = localStorage.getItem("sessionId");
      if (sessionId) {
        // Fetch user data using the session
        fetchUserData(sessionId);
      } else {
        navigate("/login");
      }
    }
    // eslint-disable-next-line react-hooks/exhaustive-deps
  }, [userData, navigate]);

  const fetchUserData = async (sessionId) => {
    try {
      const response = await fetch('http://localhost:5000/api/auth/check-session', {
        headers: {
          'X-Session-ID': sessionId
        }
      });
      const data = await response.json();
      if (data.valid) {
        // Store the user data into local component state so the dashboard can render.
        // Ensure shape matches existing usage: { user: { ... } }
        setUserData({ user: data.user });
      } else {
        navigate("/login");
      }
    } catch (error) {
      console.error("Error fetching user data:", error);
      navigate("/login");
    }
  };

  if (!userData) {
    return (
      <div className="flex items-center justify-center min-h-screen">
        <div className="text-white">Loading...</div>
      </div>
    );
  }

  return (
    <div className="min-h-screen bg-zinc-900 text-white p-8">
      <h1 className="text-3xl font-bold mb-6">Host Dashboard</h1>
      <div className="grid grid-cols-1 md:grid-cols-2 lg:grid-cols-3 gap-6">
        <div className="bg-zinc-800 p-6 rounded-lg">
          <h2 className="text-xl font-semibold mb-4">Welcome, {userData.user.name}!</h2>
          <p className="text-gray-300">Email: {userData.user.email}</p>
          <p className="text-gray-300">Organization: {userData.user.organizationName}</p>
          <p className="text-gray-300">Organization Code: {userData.user.organizationCode}</p>
        </div>
        
        <div className="bg-zinc-800 p-6 rounded-lg">
          <h3 className="text-lg font-semibold mb-2">Quick Stats</h3>
          <p className="text-gray-300">Your hosting statistics will appear here</p>
        </div>
        
        <div className="bg-zinc-800 p-6 rounded-lg">
          <h3 className="text-lg font-semibold mb-2">Recent Activity</h3>
          <p className="text-gray-300">Your recent quiz activities</p>
        </div>
      </div>
    </div>
  );
};

export default HostDashboard;
